27 inline std::vector<Batch> createBatches(ptrdiff_t begin,
28  size_t nTasks,
29  size_t nThreads,
30  size_t nBatches)
31 {
32  if (nTasks == 0)
33  return {Batch{0, 0}};
34  nThreads = std::max(nThreads, static_cast<size_t>(1));
35  if (nBatches == 0)
36  nBatches = computeNumBatches(nTasks, nThreads);
37  nBatches = std::min(nTasks, nBatches);
38  std::vector<Batch> batches(nBatches);
39  size_t minSize = nTasks / nBatches;
40  ptrdiff_t remSize = nTasks % nBatches;
41 
42  for (size_t i = 0, k = 0; i < nTasks; k++) {
43  ptrdiff_t bBegin = begin + i;
44  ptrdiff_t bSize = minSize + (remSize-- > 0);
45  batches[k] = Batch{bBegin, bBegin + bSize};
46  i += bSize;
47  }
48 
49  return batches;
50 }
