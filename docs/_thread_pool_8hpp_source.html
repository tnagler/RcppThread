28  ThreadPool(size_t nThreads) : stopped_(false)
29  {
30  for (size_t t = 0; t < nThreads; t++) {
31  pool_.emplace_back([this] {
32  // observe thread pool as long there are jobs or pool has been
33  // stopped
34  while (!stopped_ | !jobs_.empty()) {
35  std::function<void()> job;
36  {
37  // thread must hold the lock while modifying shared
38  // variables
39  std::unique_lock<std::mutex> lk(m_);
40 
41  // wait for new job or stop signal
42  cv_.wait(lk, [this] {
43  return stopped_ || !jobs_.empty();
44  });
45 
46  // check if interrupted
47  checkUserInterrupt();
48 
49  // check if there are any jobs left in the queue
50  if (jobs_.empty())
51  continue;
52 
53  // take job from the queue
54  job = std::move(jobs_.front());
55  jobs_.pop();
56  }
57 
58  // execute job
59  job();
60  }
61  }
62  );
63  }
64  }

82  auto push(F&& f, Args&&... args) -> std::future<decltype(f(args...))>
83  {
84  // create pacakged task on the heap to avoid stack overlows.
85  auto job = std::make_shared<std::packaged_task<decltype(f(args...))()>>(
86  [&f, &args...] { return f(args...); }
87  );
88 
89  // add job to the queue
90  {
91  std::unique_lock<std::mutex> lk(m_);
92  if (stopped_)
93  throw std::runtime_error("cannot push to stopped thread pool");
94  jobs_.emplace([job] () { (*job)(); });
95  }
96 
97  // signal a waiting worker that there's a new job
98  cv_.notify_one();
99 
100  // return future result of the job
101  return job->get_future();
102  }

105  void join()
106  {
107  // signal all threads to stop
108  {
109  std::unique_lock<std::mutex> lk(m_);
110  stopped_ = true;
111  }
112  cv_.notify_all();
113 
114  // join threads if not done already
115  if (pool_[0].joinable()) {
116  for (auto &worker : pool_) {
117  worker.join();
118  }
119  }
120  }
