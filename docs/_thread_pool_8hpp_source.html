29  ThreadPool(size_t nThreads) : num_busy_(0), stopped_(false)
30  {
31  for (size_t t = 0; t < nThreads; t++) {
32  pool_.emplace_back([this] {
33  // observe thread pool as long there are jobs or pool has been
34  // stopped
35  while (!stopped_ | !jobs_.empty()) {
36  // thread must hold the lock while modifying shared
37  // variables
38  std::unique_lock<std::mutex> lk(m_);
39 
40  // wait for new job or stop signal
41  cv_tasks_.wait(lk, [this] {
42  return stopped_ || !jobs_.empty();

44
