116 template<class F, class... Args>
117 void ThreadPool::push(F&& f, Args&&... args)
118 {
119  if (workers_.size() == 0) {
120  f(args...); // if there are no workers, do the job in the main thread
121  return;
122  } else {
123  // must hold lock while modifying the shared queue
124  std::lock_guard<std::mutex> lk(mTasks_);
125  if (stopped_)
126  throw std::runtime_error("cannot push to joined thread pool");
127  jobs_.emplace([f, args...] { f(args...); });
128  }
129  // signal a waiting worker that there's a new job
130  cvTasks_.notify_one();
131 }

238 inline void ThreadPool::wait()
239 {
240  while (true) {
241  if ( waitForWakeUpEvent() ) {
242  if ( !this->allJobsDone() ) {
243  this->clear(); // cancel all remaining jobs
244  continue; // wait for currently running jobs
245  }

275 inline void ThreadPool::startWorker()
276 {
277  workers_.emplace_back([this] {
278  std::function<void()> job;
279  // observe thread pool; only stop after all jobs are done
280  while (!stopped_ | !jobs_.empty()) {
281  // must hold a lock while modifying shared variables
282  std::unique_lock<std::mutex> lk(mTasks_);
283 
284  // thread should wait when there is no job
285  cvTasks_.wait(lk, [this] {
286  return stopped_ || !jobs_.empty();
287  });
288 
289  // queue can be empty if thread pool is stopped
290  if (jobs_.empty())
291  continue;
292 
293  // take job from the queue
294  job = std::move(jobs_.front());
295  jobs_.pop();
296 
297  // lock can be released before starting work, but must signal
298  // that thread will be busy before (!) to avoid premature breaks
299  this->announceBusy();
300  lk.unlock();
301 
302  this->doJob(std::move(job));
303  this->announceIdle();
304  std::this_thread::yield();
305  }
306  });
307 }

373 inline bool ThreadPool::waitForWakeUpEvent()
374 {
375  static auto timeout = std::chrono::milliseconds(250);
376  auto isWakeUpEvent = [this] {
377  return this->allJobsDone() | this->hasErrored();
378  };
379  std::unique_lock<std::mutex> lk(mTasks_);
380  cvBusy_.wait_for(lk, timeout, isWakeUpEvent);
381  return isWakeUpEvent();
