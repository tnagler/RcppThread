84 inline ThreadPool::ThreadPool(size_t nThreads)
85 {
86  for (size_t t = 0; t < nThreads; ++t) {
87  workers_.emplace_back([this] {
88  std::function<void()> job;
89  // observe thread pool; only stop after all jobs are done
90  while (!stopped_ | !jobs_.empty()) {
91  // must hold a lock while modifying shared variables
92  std::unique_lock<std::mutex> lk(mTasks_);
93 
94  // thread should wait when there is no job
95  cvTasks_.wait(lk, [this] {
96  return stopped_ || !jobs_.empty();
97  });
98 
99  // queue can be empty if thread pool is stopped
100  if (jobs_.empty())
101  continue;
102 
103  // take job from the queue
104  job = std::move(jobs_.front());
105  jobs_.pop();
106 
107  // lock can be released before starting work
108  lk.unlock();
109  this->doJob(std::move(job));
110  }
111  });
112  }
113 }

131 template<class F, class... Args>
132 auto ThreadPool::push(F&& f, Args&&... args)
133  -> std::shared_future<decltype(f(args...))>
134 {
135  using result_t = decltype(f(args...));
136  using jobPackage = std::packaged_task<result_t()>;
137 
138  // create packaged task on the heap
139  auto job = std::make_shared<jobPackage>([&f, args...] {
140  return f(args...);
141  });
142  std::shared_future<result_t> sf(std::move(job->get_future()));
143 
144  // if there are no workers, just do the job in the main thread
145  if (workers_.size() == 0) {
146  (*job)();
147  sf.get();
148  return sf;
149  }
150 
151  // add job to the queue
152  {
153  std::lock_guard<std::mutex> lk(mTasks_);
154  if (stopped_)
155  throw std::runtime_error("cannot push to joined thread pool");
156  jobs_.emplace([job] () { (*job)(); });
157  }
158 
159  // signal a waiting worker that there's a new job
160  cvTasks_.notify_one();
161 
162  // return future result of the job
163  return sf;
164 }

251 inline void ThreadPool::wait()
252 {
253  if (workers_.size() == 0) {
254  // all jobs have been executed in the main thread, no need to wait
255  checkUserInterrupt();
256  return;
257  }
258 
259  auto allJobsDone = [this] { return (numBusy_ == 0) && jobs_.empty(); };
260  auto timeout = std::chrono::milliseconds(250);
261  while (true) {
262  {
263  // wait_for tries acquires lk when waking up
264  std::unique_lock<std::mutex> lk(mTasks_);
265  cvBusy_.wait_for(lk, timeout, allJobsDone);
266  // lk can be released immediately
267  }
268 
269  // relase messages stored by worker threads to the R console
270  Rcout << "";
271 
272  // check whether timeout was reached or another event caused wake-up
273  if ( isInterrupted() ) {
274  this->clear(); // cancel all remaining jobs
275  break;
276  } else if ( allJobsDone() ) {
277  break;
278  }
279 
280  // give other threads priority before waiting again
281  std::this_thread::yield();
282  }
283 
284  // synchronize one last time before continuing main thread
285  Rcout << "";
286  checkUserInterrupt();
287 }
